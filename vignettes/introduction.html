Class definitions

library(R6)
ClassA <- R6::R6Class("ClassA",
  public = list(
    x_1 = letters,
    x_2 = TRUE,
    foo = function() {
      "foo"
    }
  ),
  private = list(
    .x_1 = LETTERS,
    .x_2 = FALSE,
    .foo = function() {
      ".foo"
    }
  )
)
ClassB <- R6::R6Class("ClassB", inherit = ClassA, public = list(x_3 = 1))
